    // Aquí es donde se establece el evento de clic
    dia.addEventListener('click', () => {
      // Comprobamos si el día que se ha pulsado es menor o igual al total de frases disponibles
      if (diaGlobal <= frases.length) {
        fraseEl.textContent = frases[diaGlobal - 1]; // Mostramos la frase correspondiente al día
        dia.classList.add('visto'); // Marcamos el día como "visto"
        diaGlobal++; // Ahora incrementamos el día global, porque ya hemos mostrado la frase para este día
      } else {
        fraseEl.textContent = "No hay frase para este día 😅"; // Si no hay más frases disponibles
      }
    });

    diasDiv.appendChild(dia);
  }

  mesDiv.appendChild(diasDiv);
  calendario.appendChild(mesDiv);
});
